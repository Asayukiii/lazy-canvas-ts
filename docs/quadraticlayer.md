## Example

```js
const { LazyCanvas, QuadraticLayer } = require('@hitomihiumi/lazy-canvas')

let quadratic = new QuadraticLayer()
.setColor('#fff')
.setStroke(5)
.setPoints({ x: 10, y: 10 }, { x: 300, y: 200 })
.setControlPoint({ x: 200, y: 100 })

const lazy = new LazyCanvas()
.addLayers(quadratic)
//...
```
